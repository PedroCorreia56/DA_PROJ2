80 int FindPath(Graph& graph,int start,int end, int capacity){
81 
82  //Vê se os valores dos vértices são válidas
83  if(start <=0 || start>graph.getNumNodes() || end <=0 || end>graph.getNumNodes() || start==end)
84  return 0;
85 
86  //Grafo Residual
87  vector<vector<int>>gf( graph.getNumNodes()+1 , vector<int> (graph.getNumNodes()+1,0));
88 
89  //Copia o grafo original para o Grafo Residual
90  for (int i = 1; i <=graph.getNumNodes() ; i++){
91  for (auto e : graph.nodes[i].adj) {
92  gf[i][e.dest]=e.cap-e.flux; //o fluxo inicial é 0, mas está assim para ser capaz de suportar o cenário 2.2
93  }
94  }
95 
96  //Vetor para guardar o caminho usado
97  vector<int> parent(graph.getNumNodes()+1, 0);
98 
99 
100  int v,u;//variavéis de apoio
101  int max_flow = 0;//Guarda o fluxo máximo
102  int used_flow=0;//Guarda o fluxo utilizado
103  while (Graph::bfs2(gf,start,end,parent,graph.getNumNodes()+1)) {
104 
105  //Guarda a capacidade residual mínima do caminho
106  int path_flow = INT_MAX;
107  //Encontra a capacidade residual mínima do caminho encontrado no BFS
108  for (v = end; v != start; v = parent[v]) {
109  u = parent[v];
110  path_flow = min(path_flow, gf[u][v]);
111  }
112 
113  //variável auxiliar
114  int added_flow;
115  //Caso o fluxo máximo+fluxo novo caminho seja menor que o capacidade dada
116  if((max_flow+path_flow)<=capacity)
117  added_flow=path_flow;
118  else//Caso a soma do fluxo do novo caminho com o fluxo máximo seja superior à capacidade, apenas usa-se a quantidade necessária
119  added_flow=capacity-max_flow;
120 
121  //Atualiza a capacidade residual das arestas normais e
122  // das arestas inversas ao longo do caminho
123  for (v = end; v != start; v = parent[v]) {
124  u = parent[v];
125  gf[u][v] -= added_flow;
126  gf[v][u] += added_flow;
127  }
128  //Se isto acontecer quer dizer que já se chegou à capacidade pedida
129  if(added_flow!=path_flow){
130  max_flow += path_flow;
131  used_flow+=added_flow;
132  break;
133  }
134 
135  //Adiciona o fluxo adicionado ao fluxo usado
136  used_flow+=added_flow;
137  //adiciona o fluxo do caminho encontrado ao fluxo total
138  max_flow += path_flow;
139  }
140 
141 
142  // Atualiza o fluxo das arestas do grafo original
143  for (int i = 1; i <=graph.getNumNodes() ; i++){
144  for (auto & e : graph.nodes[i].adj) {
145  //Caso a aresta não tenha fluxo inicial
146  if (e.flux==0)
147  e.flux+=(e.cap-gf[i][e.dest]);
148  else //Caso a aresta tenha fluxo inical, usado por causa do cenário 2.2
149  e.flux+=((e.cap-e.flux)-gf[i][e.dest]);
150  }
151  }
152 
153  cout<<"Tamanho do grupo ''usado''"<<used_flow<<endl;
154  cout<<"Tamanho do grupo máximo:"<<max_flow<<endl;
155  if(used_flow!=0)
156  return used_flow;
157  else
158  return max_flow;
159

208 
209  for (int i = start; i <=graph.getNumNodes() ; i++) {
