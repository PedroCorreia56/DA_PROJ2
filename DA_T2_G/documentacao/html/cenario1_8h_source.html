39 static pair<int,vector<int>> MaxGroupSize(Graph graph,int start, int end){
40 
41  //Array que guarda a máxima capaciade mínima de cada vértice
42  vector<int> capacidade(graph.getNumNodes()+1, INT_MIN);
43 
44  //Vetor para guardar o caminho usado
45  vector<int> parent(graph.getNumNodes()+1, 0);
46 
47  //heap para guardar no topo o vértice com maior capacidade
48  MaxHeap<int,int> heap(graph.getNumNodes(),-1);
49  for (int i = 1; i <= graph.getNumNodes(); i++) { // O(n)
50  heap.insert(i,capacidade[i]);
51  }
52  capacidade[start]=INT_MAX;
53  heap.increaseKey(start,INT_MAX);
54 
55  //Descobre máxima capaciade mínima de cada vértice
56  while(heap.getSize()!=0){
57  int t= heap.removeMax(); //O(n)
58  for (auto w : graph.nodes[t].adj) { // O(n)
59  if(min(capacidade[t],w.cap)>capacidade[w.dest]){
60  capacidade[w.dest]=min(capacidade[t],w.cap);
61  parent[w.dest]=t;
62  heap.increaseKey(w.dest,capacidade[w.dest]); // O(n)
63  }
64  }
65  }
66 
67  //Caso de erro
68  if(capacidade[end]==INT_MAX || capacidade[end]==INT_MIN){
69  return {0,parent} ;
70  }
71  return {capacidade[end],parent};
72 }
